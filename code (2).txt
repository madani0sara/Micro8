#include <Keypad.h>
const byte ROWNS = 4;
const byte COLS = 4;
char keys[ROWNS][COLS] = {
  {'1', '2', '3', 'A'},
  {'4', '5', '6', 'B'},
  {'7', '8', '9', 'C'},
  {'*', '0', '#', 'D'}
};
byte rowPins [ROWNS] = {2,3,4,5};
byte colPins [COLS] = {6,7,8,9};
Keypad kpd = Keypad(makeKeymap(keys), rowPins, colPins, ROWNS, COLS);

void setup() {
  Serial.begin(9600);
}

void loop() {
  char key = kpd.getKey();
  if (key) {
    Serial.print(key);
  }
}
